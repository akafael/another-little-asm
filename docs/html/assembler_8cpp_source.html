375 bool addNewSymbolINST1(string strInst1,string strArg1, int lineCount,string line)
376 {
377  int code = isValidInstructionCall(strInst1,strArg1);
378  if(code==INVALID_INSTRUCTION)
379  {

383  {
384  PRINT_ERR_ARG_NUM(lineCount,line);
385  }
386  else if(code==INVALID_ARG1)
387  {
388  PRINT_ERR_ARG(lineCount,line,strArg1);
389  }
390  else if(code<=14)// Chamada Válida de instrução
391  {
392  // Cria Simbolos e coloca na tabela
393  symbol tmp_symb0,tmp_symb1;
394  int labelPos;
395 
396  tmp_symb0.type = SYM_INSTRUCTION;
397  tmp_symb0.content = code;
398  tmp_symb0.address = currentSymbolAddr;
399  tmp_symb0.line = lineCount;
400  symbolsTable.push_back(tmp_symb0);
401 
402  tmp_symb1.type = SYM_LABEL;
403  tmp_symb1.address = currentSymbolAddr+1;
404  tmp_symb1.text = strArg1;
405  tmp_symb1.line = lineCount;
406 
407  labelPos = findLabel(labelsTable,strArg1);
408  if(labelPos==LABEL_NOT_FOUND)
409  {
410  tmp_symb1.content = UNDEFINED_LABEL_ADDR;
411  /// @todo Modificar para permitir completar tabela ao final da passagem
412  }
413  else
414  {
415  tmp_symb1.content = labelsTable[labelPos].addr;
416 
417  // Verifica Erro de Divisão por zero
418  if((code==DIV)&&(symbolsTable[tmp_symb1.content].content==0))
419  {
420  PRINT_ERR_DIV0(lineCount,line,strArg1);
421  }
422 
423  // Verifica erro de escrita em constante
424  if((code==STORE||code==INPUT)\

426  {
427  PRINT_ERR_ARG_TYPE_CONST(lineCount,line,strArg1);
428  }

433  // Atualiza Endereço p/ próximo simbolo
434  currentSymbolAddr+=2;
435 
436  // Verifica se Instrução está na sessão certa
437  if(currentSection!=SECTION_TEXT)
438  PRINT_ERR_WRONG_SECTION_TEXT_INSTRUCTION(lineCount,line);
439 
440  }
441  else
442  {
443  // Possível Diretiva
444  switch (code) {
445  case SECTION:
446  if(strArg1=="TEXT")

465  {
466  symbol tmp_symb0;
467 
468  tmp_symb0.type = SYM_NUM_DEC;

470  tmp_symb0.address = currentSymbolAddr+i;
471  tmp_symb0.line = lineCount;
472  symbolsTable.push_back(tmp_symb0);
473  }
474  currentSymbolAddr+=spaceAmount; // Atualiza Endereços
475 
476  // Verifica se a Diretiva está na sessão certa
477  if(currentSection!=SECTION_DATA)
478  PRINT_ERR_WRONG_SECTION_DATA_INSTRUCTION(lineCount,line);
479  break;
480  }
481  case CONST:
482  // Erro
483  PRINT_ERR_ARG(lineCount,line,strArg1);
484 
485  // Verifica se a Diretiva está na sessão certa
486  if(currentSection!=SECTION_DATA)
487  PRINT_ERR_WRONG_SECTION_DATA_INSTRUCTION(lineCount,line);
